#include<iostream>
#include<string>

using namespace std;

int main(){
    string s1,s2;
    while(getline(cin,s1) && getline(cin,s2)){
        int hashtable[256] = {0};
        for(size_t i = 0;i < s2.size();i++){
            hashtable[s2[i]] = 1;
        }
        string ret;
        for(size_t j = 0;j < s1.size();j++){
            if(hashtable[s1[j]] != 1){
                ret += s1[j];
            }
        }
        cout << ret << endl;
    }
    return 0;
}
